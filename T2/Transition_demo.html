Transition:

1. transition-property: all/specific;
2. transition-duration: tms;
3. transition-timing-function:linear|ease-in|ease-out|ease;
    -- ease-in: starts slower and ends faster
    -- ease-out: starts faster and ends slower
    -- ease-in-out: starts slower and ends slower
4. transition-delay:1s;

-- shorthand: transition: property duration function delay
