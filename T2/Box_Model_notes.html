1. border-style:
            i. border-top-style: solid|dashed|dotted;
            ii. border-right-style: double|groove|ridge;
            iii. border-bottom-style: inset|outset|none;
            iv. border-left-style: hidden;
            --none: border will neither take space nor be created
            --hidden: border will take space but won't be created
            --shorthand: 
                border-style: style; (all sides are dashed) 
                border-style: style1 style2; (style1: top and bottom, style2: left and right) 
                border-style: style1 style2 style3; (style1: top, style2: left and right, style3: bottom) 
                border-style: style1 style2 style3 style4; (style1: top, style2: right, style3: bottom, style4: left) 
        
        2. border-width:
            --shorthand: 
                border-width: width; (all sides are dashed) 
                border-width: width1 width2; (width1: top and bottom, width2: left and right) 
                border-width: width1 width2 width3; (width1: top, width2: left and right, width3: bottom) 
                border-width: width1 width2 width3 width4; (width1: top, width2: right, width3: bottom, width4)
        
        3. border-color:
            --shorthand: 
                border-color: color; (all sides are dashed) 
                border-color: color1 color2; (color1: top and bottom, color2: left and right) 
                border-color: color1 color2 color3; (color1: top, color2: left and right, color3: bottom) 
                border-color: color1 color2 color3 color4; (color1: top, color2: right, color3: bottom, color4)
