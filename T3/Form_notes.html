        2.  Inline Form (inline form) (using grid system)
    
        
Stacked Form:

            Label -> class="form-label"
            input/textarea -> class="form-control" (box with borders)
            input type="text" -> class="form-control-plaintext" (box without borders)
            input type="color" -> class="form-control form-control-color"
            input type="range" -> class="form-range" (by default no color in slider, provide width in style for color)
            select -> class="form-select"
            1 (radio/checkbox) -> Inside a div : class="form-check" 
                                     checkbox/radio : class="form-check-input"
                                     provide name and id
                                     label : class="form-check-label"
